float pp1 =0;
float pp2 =0;
int hp1 ;
int hp2 ;
int e1  ;
int e2  ;

int p1at1d;
int p1at1c;

int p2at1d;
int p2at1c;

int p1at2d;
int p1at2c;

int p2at2d;
int p2at2c;

int game(int h1,int h2,int n1, int n2,int i){
 
    int hpp,nn;

    if (i%2)
    {
        hpp = h1-p2at1d;
        nn = n2-p2at1c;
        if (nn<=0)
        {
            return 0;
        }
        
        if (hpp<=0)
        {
            pp2++;
            return 0;
        }
        else{
            game(hpp,h2,n1,nn,++i);
        }


        hpp = h1-p2at2d;
        nn = n2-p2at2c;
        if (nn<=0)
        {
            return 0;
        }
        
        if (hpp<=0)
        {
            pp2++;
            return 0;
        }
        else{
            game(hpp,h2,n1,nn,++i);
        }
    }

    else{
        hpp = h2-p1at1d;
        nn = n1-p1at1c;
        if (nn<=0)
        {
            return 0;
        }
        
        if (hpp<=0)
        {
            pp1++;
            return 0;
        }
        else{
            game(h1,hpp,nn,n2,++i);
        }
        hpp = h2-p1at2d;
        nn = n1-p1at2c;
        if (nn<0)
        {
            return 0;
        }
        
        if (hpp<0)
        {
            pp1++;
            return 0;
        }
        else{
            game(h1,hpp,nn,n2,++i);
        }
    }

            
    return 1;
}







int main(){

    int i =0;
    

    game(hp1,hp2,e1,e2,i);

    cout<<pp1;//porazi

}
